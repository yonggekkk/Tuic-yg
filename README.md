```
bash <(wget -qO- https://gitlab.com/rwkgyg/tuic-yg/raw/main/tuic.sh 2> /dev/null)
```

```
bash <(curl -Ls https://gitlab.com/rwkgyg/tuic-yg/raw/main/tuic.sh)
```
